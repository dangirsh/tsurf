    **Test 8 — graceful shutdown drains in-flight request (DEG-04):**
    ```rust
    #[tokio::test]
    async fn test_graceful_shutdown_drains_request() {
        use tokio::sync::oneshot;

        // Mock upstream: hold response for 200ms then reply
        let (upstream_listener, upstream_port) = random_listener().await;
        tokio::spawn(async move {
            axum::serve(upstream_listener, Router::new().route("/slow", get(|| async {
                tokio::time::sleep(std::time::Duration::from_millis(200)).await;
                "done"
            }))).await.unwrap();
        });

        // Proxy with graceful shutdown channel
        let (proxy_listener, proxy_port) = random_listener().await;
        let state = make_state("sk-ant-api03-placeholder", upstream_port, "real-key");
        let (shutdown_tx, shutdown_rx) = oneshot::channel::<()>();
        tokio::spawn(async move {
            axum::serve(proxy_listener, make_router(state))
                .with_graceful_shutdown(async { shutdown_rx.await.ok(); })
                .await
                .unwrap();
        });

        // Start a slow request in a background task
        let proxy_port_clone = proxy_port;
        let upstream_port_clone = upstream_port;
        let req_task = tokio::spawn(async move {
            Client::new()
                .get(format!("http://127.0.0.1:{}/slow", proxy_port_clone))
                .header("Host", format!("127.0.0.1:{}", upstream_port_clone))
                .header("x-api-key", "sk-ant-api03-placeholder")
                .send().await
        });

        // Give the request 50ms to reach the upstream, then trigger shutdown
        tokio::time::sleep(std::time::Duration::from_millis(50)).await;
        shutdown_tx.send(()).unwrap();

        // The in-flight request should still complete
        let resp = req_task.await.unwrap().unwrap();
        assert_eq!(resp.status(), 200);
        assert_eq!(resp.text().await.unwrap(), "done");
    }
    ```
