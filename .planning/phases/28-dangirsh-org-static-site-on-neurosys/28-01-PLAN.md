**Step 1: Assess Haskell compatibility**

The site.cabal declares:

- `pandoc >= 3.0 && < 3.6`
- `pandoc-types >= 1.23`

Check what nixpkgs-25.11 provides:


If hakyll is >= 4.17, widen the upper bound in site.cabal to `< 4.18` (or remove it). Similarly for pandoc if it's >= 3.6.

**Step 2: Create flake.nix**

```nix
{
  description = "dangirsh.org - personal website built with Hakyll";

  inputs = {
    nixpkgs.url = "github:NixOS/nixpkgs/nixos-25.11";
  };

  outputs = { self, nixpkgs }:
    let
      system = "x86_64-linux";
      pkgs = nixpkgs.legacyPackages.${system};

      # Build the Hakyll generator executable
      generator = pkgs.stdenv.mkDerivation {
        name = "dangirsh-generator";
        src = ./generator;
        phases = "unpackPhase buildPhase";
        buildInputs = [
          (pkgs.haskellPackages.ghcWithPackages (p: with p; [
            hakyll
            pandoc
            pandoc-types
            containers
          ]))
        ];
        buildPhase = ''
          mkdir -p $out/bin
          ghc -O2 -dynamic --make site.hs -o $out/bin/generator
        '';
      };

      # Build the static site using the generator
      site = pkgs.stdenv.mkDerivation {
        name = "dangirsh-org-site";
        src = ./site;
        phases = "unpackPhase buildPhase";
        nativeBuildInputs = [ generator ];
        LANG = "en_US.UTF-8";
        LOCALE_ARCHIVE = "${pkgs.glibcLocales}/lib/locale/locale-archive";
        buildPhase = ''
          generator build
          mkdir $out
          cp -r _site/* $out
        '';
      };
    in {
      packages.${system}.default = site;
    };
}
```

Key design choices:
- Pin to `nixos-25.11` to match neurosys (consumer can override with `follows`)
- Expose `packages.x86_64-linux.default` (standard flake convention)
- Set `LOCALE_ARCHIVE` for Hakyll's UTF-8 requirements (the old build used `LANG=en_US.UTF-8`)
- Use `haskellPackages.ghcWithPackages` to match the approach from `default.nix`

**Step 3: Handle Haskell build failures**

If compilation fails due to version bounds:
1. Widen `site.cabal` bounds to match available versions
2. If specific packages fail, try `haskell.lib.dontCheck` on them:
   ```nix
   haskellPackages.override {
     overrides = self: super: {
       hakyll = pkgs.haskell.lib.dontCheck super.hakyll;
     };
   }
   ```
3. As a last resort, use `haskellPackages.callCabal2nix` to build from source with custom bounds
