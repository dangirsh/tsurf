**Create `modules/nginx.nix`:**

```nix
# modules/nginx.nix
# @decision WEB-01: nginx as unified reverse proxy for all public web traffic (dangirsh.org + claw-swap.com)
# @rationale: Single internet-facing process minimizes attack surface. Docker Caddy removed per SECURITY-01.
# @decision WEB-02: ACME via Let's Encrypt for TLS — no Cloudflare origin certs, no manual cert management
# @rationale: NixOS-native ACME handles renewal automatically. Cloudflare certs were Caddy-specific.
# @decision WEB-03: dangirsh.org served as static files from Nix store path (no reverse proxy)
# @rationale: Hakyll output is a Nix derivation; nginx serves it directly. Zero runtime dependencies.
# @decision WEB-04: www.dangirsh.org redirects to dangirsh.org (canonical domain, no www)
# @decision WEB-05: Static asset cache headers for performance (CSS/JS/images get long cache)
{ config, inputs, lib, pkgs, ... }:
let
  # dangirsh-site flake produces the static site as packages.x86_64-linux.default
  siteRoot = inputs.dangirsh-site.packages."x86_64-linux".default;
in {
  # ── ACME (Let's Encrypt) ─────────────────────────────────────────────
  security.acme = {
    acceptTerms = true;
    defaults.email = "dan@dangirsh.org";
    # @decision WEB-06: Use production ACME server. Switch to staging for testing:
    # defaults.server = "https://acme-staging-v02.api.letsencrypt.org/directory";
  };

  # ── nginx ────────────────────────────────────────────────────────────
  services.nginx = {
    enable = true;
    # openFirewall = false: ports 80/443 are already in networking.nix allowedTCPPorts
    # Recommended security settings
    recommendedTlsSettings = true;
    recommendedOptimisation = true;
    recommendedGzipSettings = true;
    recommendedProxySettings = true;

    # ── dangirsh.org (static site) ────────────────────────────────────
    virtualHosts."dangirsh.org" = {
      enableACME = true;
      forceSSL = true;
      root = siteRoot;

      locations."/" = {
        tryFiles = "$uri $uri/index.html $uri.html =404";
      };

      # Static asset caching (CSS, JS, images, fonts)
      locations."~* \\.(css|js|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$" = {
        root = siteRoot;
        extraConfig = ''
          expires 30d;
          add_header Cache-Control "public, immutable";
        '';
      };
    };

    # ── www.dangirsh.org -> dangirsh.org redirect ─────────────────────
    virtualHosts."www.dangirsh.org" = {
      enableACME = true;
      forceSSL = true;
      globalRedirect = "dangirsh.org";
    };

    # ── claw-swap.com (reverse proxy to app container) ────────────────
    virtualHosts."claw-swap.com" = {
      enableACME = true;
      forceSSL = true;
      locations."/" = {
        proxyPass = "http://127.0.0.1:3000";
        proxyWebsockets = true;
      };
    };
  };
}
```
