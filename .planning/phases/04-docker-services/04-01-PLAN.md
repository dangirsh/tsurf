  <action>
    Create `nix/claw-swap-app.nix` that builds the claw-swap Hono app as a Docker image using `buildNpmPackage` + `dockerTools.buildLayeredImage`.

    **Pattern:** Same as parts-agent.nix but simpler (no workspace complexity, single package.json).

    ```nix
    # nix/claw-swap-app.nix -- Docker image for claw-swap Hono API server
    #
    # @decision: Use buildNpmPackage directly (single package.json, no workspaces).
    # @decision: sharp is unused in source (zero imports) -- use --ignore-scripts to skip its native addon download.
    #   If sharp is added to source later, add pkgs.vips to nativeBuildInputs and run npm rebuild sharp in buildPhase.
    { pkgs, src, ... }:

    let
      nodejs = pkgs.nodejs_22;

      claw-swap-app = pkgs.buildNpmPackage {
        pname = "claw-swap";
        version = "0.1.0";
        inherit src;

        npmDepsHash = "";  # PLACEHOLDER -- compute after first build attempt

        inherit nodejs;

        # sharp v0.34.5 downloads prebuilt libvips binaries in postinstall -- blocked in Nix sandbox.
        # sharp has zero imports in src/ (confirmed grep), so we just skip its install script.
        npmFlags = [ "--ignore-scripts" ];
        makeCacheWritable = true;

        # TypeScript compilation
        npmBuildScript = "build";

        installPhase = ''
          runHook preInstall

          mkdir -p $out/lib/claw-swap
          cp -r dist $out/lib/claw-swap/dist
          cp -r node_modules $out/lib/claw-swap/node_modules
          cp package.json $out/lib/claw-swap/
          # Static assets served by the app
          cp -r public $out/lib/claw-swap/public
          # Drizzle migrations (for manual apply via docker exec)
          cp -r drizzle $out/lib/claw-swap/drizzle 2>/dev/null || true

          runHook postInstall
        '';
      };

    in
    pkgs.dockerTools.buildLayeredImage {
      name = "claw-swap";
      tag = "latest";

      contents = with pkgs.dockerTools; [
        caCertificates
        fakeNss
      ];

      config = {
        Cmd = [ "${nodejs}/bin/node" "${claw-swap-app}/lib/claw-swap/dist/index.js" ];
        Env = [
          "NODE_ENV=production"
          "PORT=3000"
        ];
        ExposedPorts = { "3000/tcp" = {}; };
        WorkingDir = "${claw-swap-app}/lib/claw-swap";
      };
    }
    ```

    **Build steps:**
    1. Create `nix/` directory and write the file
    2. `git add nix/claw-swap-app.nix`
    3. Attempt build: `nix-build -E 'let pkgs = import <nixpkgs> {}; in pkgs.callPackage ./nix/claw-swap-app.nix { src = ./.; }'` -- this will fail on npmDepsHash
    4. Compute the correct hash from the error output (the error message shows the expected hash)
    5. Update npmDepsHash with the correct value
    6. Rebuild -- should succeed

    **If sharp causes issues beyond --ignore-scripts:** The app does not import sharp. If the build still pulls in sharp's native binaries, add `npmFlags = [ "--ignore-scripts" "--omit=optional" ]` or patch package.json in `postPatch` to remove sharp from dependencies.

    **IMPORTANT:** The build may take several minutes as it downloads npm dependencies and compiles TypeScript. Be patient.
