**Step 1: Move host-specific settings OUT of shared modules**

Four modules have hardcoded host-specific values that must be parameterized:

1. **`modules/secrets.nix`**: Remove `sops.defaultSopsFile = ../secrets/neurosys.yaml;`. Move it to each host's `default.nix`. Keep the `sops.age.sshKeyPaths` and all secret declarations in the shared module (they're host-agnostic — they just declare what secrets exist, the defaultSopsFile determines which encrypted file to read).

2. **`modules/docker.nix`**: Replace `externalInterface = "eth0";` with a value that comes from the host config. Use `config.networking.nat.externalInterface` already being set — actually, the simplest approach is to set `networking.nat.externalInterface` in each host's `default.nix` and remove it from `docker.nix`. The NAT `enable = true` and `internalIPs` stay in docker.nix (shared).

3. **`modules/homepage.nix`**: Multiple values are hardcoded to "neurosys":
   - **`tailscaleIP`**: Replace the `let tailscaleIP = "100.127.245.9"` approach with `allowedHosts = "*"` — this is safe because homepage is only accessible via `trustedInterfaces = ["tailscale0"]` anyway.
   - **`title`**: Replace any hardcoded `"neurosys"` dashboard title with `config.networking.hostName` so OVH shows "neurosys-prod".
   - **`openmeteo label`**: Replace any hardcoded "neurosys" location/label in the openmeteo widget with `config.networking.hostName`.
   - **`allowedHosts`**: Remove any hardcoded "neurosys" from allowedHosts — the wildcard `"*"` handles both hosts.
   - **Home Assistant `href`**: Replace any hardcoded `http://neurosys:8123` or `http://100.127.245.9:8123` href with `http://${config.networking.hostName}:8123` so it resolves via Tailscale MagicDNS on each host.

4. **`modules/boot.nix`**: Remove `device = "/dev/sda";` from the shared boot module. Move `boot.loader.grub.device` to each host's `default.nix`. Keep the GRUB settings (enable, efiSupport, efiInstallAsRemovable, configurationLimit) and the initrd rollback script in the shared module (they use `disk-main-root` partlabel which is set by disko and is the same for both hosts).

**Step 1: Refactor flake.nix**

Extract the shared module list into a `commonModules` let-binding and create a `mkHost` helper function:

```nix
outputs = { self, nixpkgs, ... } @ inputs:
  let
    system = "x86_64-linux";
    pkgs = nixpkgs.legacyPackages.${system};
    treefmtEval = treefmt-nix.lib.evalModule pkgs ./treefmt.nix;

    commonModules = [
      srvos.nixosModules.server
      disko.nixosModules.disko
      impermanence.nixosModules.impermanence
      sops-nix.nixosModules.sops
      home-manager.nixosModules.home-manager
      inputs.parts.nixosModules.default
      inputs.claw-swap.nixosModules.default
      { nixpkgs.overlays = [ llm-agents.overlays.default ]; }
      {
        home-manager.useGlobalPkgs = true;
        home-manager.useUserPackages = true;
        home-manager.extraSpecialArgs = { inherit inputs; };
        home-manager.users.dangirsh = import ./home;
      }
    ];

    # IMPORTANT: ./modules is NOT in commonModules. Each host's default.nix
    # imports ../../modules in its own imports list (unchanged from today).
    # Adding ./modules to commonModules would cause a double-import.
    mkHost = hostDir: nixpkgs.lib.nixosSystem {
      inherit system;
      specialArgs = { inherit inputs; };
      modules = commonModules ++ [ hostDir ];
    };
  in {
    nixosConfigurations.neurosys = mkHost ./hosts/neurosys;
    nixosConfigurations.ovh = mkHost ./hosts/ovh;

    deploy.nodes.neurosys = {
      hostname = "neurosys";
      sshUser = "root";
      magicRollback = true;
      autoRollback = true;
      confirmTimeout = 120;
      profiles.system = {
        user = "root";
        path = deploy-rs.lib.${system}.activate.nixos self.nixosConfigurations.neurosys;
      };
    };

    deploy.nodes.ovh = {
      hostname = "neurosys-prod";  # Tailscale MagicDNS name (after Tailscale join)
      sshUser = "root";
      magicRollback = true;
      autoRollback = true;
      confirmTimeout = 120;
      profiles.system = {
        user = "root";
        path = deploy-rs.lib.${system}.activate.nixos self.nixosConfigurations.ovh;
      };
    };

    packages.${system}.deploy-rs = deploy-rs.packages.${system}.default;
    formatter.${system} = treefmtEval.config.build.wrapper;
    devShells.${system}.default = pkgs.mkShell { ... };  # Keep existing
    checks = builtins.mapAttrs ...;  # Keep existing
  };
```
